function boss_death( event )
	local caster = event.unit
	local angle = 0

	local death_dummy = CreateUnitByName("npc_dummy_unit", caster:GetAbsOrigin(), false, nil, nil, DOTA_TEAM_NEUTRALS)
	local abPass = arena_start_dummy:GetAbilityByIndex(0)
	abPass:SetLevel(1)
	Timers:CreateTimer(5, function()
		UTIL_Remove(death_dummy)
	end)

	print(boss_poss)

	for i=0, 7 do
		local effect_pos = death_dummy:GetAbsOrigin() + Vector(cos(angle), sin(angle), 0) * 400 

		print(effect_pos)
		local fire_death_effect = ParticleManager:CreateParticle("particles/fireball_explode.vpcf", PATTACH_ABSORIGIN, death_dummy)
		ParticleManager:SetParticleControl(fire_death_effect, 1, effect_pos)
		angle = angle + 45
	end
end
